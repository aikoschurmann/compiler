a :i64 = 55;


fn fold(accumulator: fn(i64, i64) -> i64,
        arr: i64[]) -> i64 
{
    len: i64 = length(arr);

    if (len == 0) {
        return 0;
    }

    result: i64 = arr[0];

    for (i: i64 = 1; i < len; i = i + 1) {
        result = accumulator(result, arr[i]);
    }

    return result;
}
